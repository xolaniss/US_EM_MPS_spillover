# South Africa
vars <- c("FFR", "FG", "LSAP", "SA_diff")
formula <- formula(value ~ SA_diff)

sa_ols_models_tbl <- 
  announcement_days_tbl |> 
  ols_model(country_var = SA_diff,
            vars = vars, 
            formula = formula,
            country_name = "South Africa")

sa_residuals_tbl <- 
  robust_residuals(announcement_days_tbl, 
                 vars = vars, 
                 country_var = SA_diff,
                 formula = formula,
                 country_replace = "SA_diff",
                 country_name = "South Africa")
