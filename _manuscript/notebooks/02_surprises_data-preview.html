surprises_wide_tbl <- read_excel(here("notebooks","data", "factors2023.xlsx")) |> 
  mutate(
    Date = as.Date(paste0(year,"-", month, "-", day))
  ) |> 
  relocate(
    Date,
    .before = year
  ) |> 
  dplyr::select(
    -c(year, month, day, `-LSAP`)
  ) 

surprises_tbl <- 
  surprises_wide_tbl |> 
  pivot_longer(
    cols = -Date,
    names_to = "Series",
    values_to = "Value"
  )
